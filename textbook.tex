public static <E> boolean isPermutation(List<E> listA, List<E> listB) {
	
	if(listA.size() != listB.size()) {
		return false;
	}
	for(int i  = 0; i < listA.size() ; i++){
		E item =  listA.get(i);
		int countA = 0;
		int countB = 0;
		
		for (E element : listA) {
			if(item.equals(element)){
				countA++;
			}
		}
		for (E element : listB) {
			if(item.equals(element)){
				countB++;
			}
		}
		if(countA != countB) {
			return false;
		}
	}
	return true;
}
